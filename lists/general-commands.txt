vim $DOTFILES_DIR/lists/general-commands.txt                                    # EDIT
sh $DOTFILES_DIR/scripts/general/copy-file-content.sh                           # COPY
echo $USER                                                                      # GET USER
uname -a                                                                        # GET SYSTEM INFORMATION
ls -l                                                                           # SHOW PERMISSIONS
source .env                                                                     # SOURCE ENV
env | cut -d'=' -f1 | fzf | sed -e 's/^/$/' -e 's/$//g' | tr -d '\n' | pbcopy   # SELECT ENV
echo -n "$(pwd)" | pbcopy                                                       # COPY DIR
basename "$PWD" | pbcopy                                                        # BASENAME
ls | fzf | tr -d '\n' | pbcopy                                                  # COPY FILENAME
top                                                                             # GET RESOURCES
chmod +x $(find . -maxdepth 1 | fzf)                                            # CHANGEMODE
realpath $(fzf) | sed "s|^$HOME|~|" | pbcopy                                    # FULL PATH
system_profiler SPSoftwareDataType SPHardwareDataType                           # INFORMATION
sed -i '' "s/TO_BE_REPLACED/REPLACEMENT/g" FILE.txt                             # SED
sh $DOTFILES_DIR/scripts/general/stop-service.sh                                # STOP SERVICE
sh $DOTFILES_DIR/scripts/general/toggle-dark-mode.sh                            # TOGGLE DARK MODE
sh $DOTFILES_DIR/scripts/general/unzip-archive.sh                               # UNZIP ARCHIVE
